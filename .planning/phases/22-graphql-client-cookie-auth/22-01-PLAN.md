    GREEN implementation:
    Create src/shared/graphql_client.ts with:
    - const GRAPHQL_ENDPOINT = "https://api.trackmangolf.com/graphql"
    - interface GraphQLResponse<T> { data: T | null; errors?: Array<{ message: string; extensions?: { code?: string } }> }
    - type AuthStatus = { kind: "authenticated" } | { kind: "unauthenticated" } | { kind: "error"; message: string }
    - async function executeQuery<T>(query: string, variables?: Record<string, unknown>): Promise<GraphQLResponse<T>>
      - Calls fetch(GRAPHQL_ENDPOINT, { method: "POST", credentials: "include", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ query, variables }) })
      - If !response.ok, throws new Error(`HTTP ${response.status}`)
      - Returns response.json() as Promise<GraphQLResponse<T>>
    - function classifyAuthResult(result: GraphQLResponse<{ me: { id: string } | null }>): AuthStatus
      - If result.errors exists and has length > 0:
        - Extract code = result.errors[0].extensions?.code ?? ""
        - Extract msg = result.errors[0].message ?? ""
        - If code === "UNAUTHENTICATED" or msg.toLowerCase() includes "unauthorized" or "unauthenticated" or "not logged in" → return { kind: "unauthenticated" }
        - Else → return { kind: "error", message: "Unable to reach Trackman — try again later" }
      - If !result.data?.me?.id → return { kind: "unauthenticated" }
      - Else → return { kind: "authenticated" }
    - const HEALTH_CHECK_QUERY = `query HealthCheck { me { id } }`
    - Export all: executeQuery, classifyAuthResult, HEALTH_CHECK_QUERY, GRAPHQL_ENDPOINT, GraphQLResponse, AuthStatus

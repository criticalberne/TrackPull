    2. Add a `showImportStatus` function (before the DOMContentLoaded handler):
    ```typescript
    /**
     * Display import result and clear IMPORT_STATUS from storage (D-02: auto-clear on read).
     * For success: show a brief success message in the portal section.
     * For error: show the error message in the portal section.
     * For importing: show an in-progress indicator.
     * For idle/absent: no-op.
     */
    function showImportStatus(status: ImportStatus): void {
      const section = document.getElementById("portal-section");
      if (!section) return;

      // Find or create the import status element within portal section
      let statusEl = document.getElementById("import-status-msg");
      if (!statusEl) {
        statusEl = document.createElement("div");
        statusEl.id = "import-status-msg";
        statusEl.style.padding = "8px 12px";
        statusEl.style.marginTop = "8px";
        statusEl.style.borderRadius = "4px";
        statusEl.style.fontSize = "13px";
        section.appendChild(statusEl);
      }

      if (status.state === "success") {
        statusEl.textContent = "Session imported successfully";
        statusEl.style.background = "var(--success-bg, #e6f4ea)";
        statusEl.style.color = "var(--success-text, #1e7e34)";
        statusEl.style.display = "block";
        // Auto-dismiss after 5 seconds
        setTimeout(() => { statusEl!.style.display = "none"; }, 5000);
      } else if (status.state === "error") {
        statusEl.textContent = status.message;
        statusEl.style.background = "var(--error-bg, #fce8e6)";
        statusEl.style.color = "var(--error-text, #c5221f)";
        statusEl.style.display = "block";
      } else if (status.state === "importing") {
        statusEl.textContent = "Importing session...";
        statusEl.style.background = "var(--info-bg, #e8f0fe)";
        statusEl.style.color = "var(--info-text, #1967d2)";
        statusEl.style.display = "block";
      } else {
        statusEl.style.display = "none";
      }
    }
    ```

    3. Inside the `DOMContentLoaded` handler, after the existing `chrome.storage.local.get` call for TRACKMAN_DATA (around line 196, after `cachedData` is set), add the import status check:
    ```typescript
    // RESIL-02: Read and display import status on popup open
    const statusResult = await new Promise<Record<string, unknown>>((resolve) => {
      chrome.storage.local.get([STORAGE_KEYS.IMPORT_STATUS], resolve);
    });
    const importStatus = statusResult[STORAGE_KEYS.IMPORT_STATUS] as ImportStatus | undefined;
    if (importStatus && importStatus.state !== "idle") {
      showImportStatus(importStatus);
      // D-02: Auto-clear on read — clear success and error states after display
      if (importStatus.state === "success" || importStatus.state === "error") {
        chrome.storage.local.remove(STORAGE_KEYS.IMPORT_STATUS);
      }
    }
    ```

    4. Also add a `storage.onChanged` listener for IMPORT_STATUS (after the existing `storage.onChanged` listener at the bottom of the DOMContentLoaded handler or in the global scope where listeners are set up) so that if the popup is still open when the import completes, it updates in real time:
    ```typescript
    // Listen for import status changes while popup is open
    chrome.storage.onChanged.addListener((changes, namespace) => {
      if (namespace === "local" && changes[STORAGE_KEYS.IMPORT_STATUS]) {
        const newStatus = changes[STORAGE_KEYS.IMPORT_STATUS].newValue as ImportStatus | undefined;
        if (newStatus && newStatus.state !== "idle") {
          showImportStatus(newStatus);
          // D-02: Auto-clear completed states
          if (newStatus.state === "success" || newStatus.state === "error") {
            chrome.storage.local.remove(STORAGE_KEYS.IMPORT_STATUS);
          }
        }
      }
    });
    ```
